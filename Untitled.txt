' Count rows where title exists but amount is missing
                If Not IsEmpty(titleVal) And Trim(CStr(titleVal)) <> "" Then
                    If IsEmpty(amountVal) Or Trim(CStr(amountVal)) = "" Or Not IsNumeric(amountVal) Then
                        missingValues = missingValues + 1
                        ' Every other missing value completes a pair — add to list on even count
                        If missingValues Mod 2 = 0 Then
                            Dim missingTitle1 As String
                            Dim missingTitle2 As String
                            missingTitle1 = Trim(CStr(wsStmt.Cells(r - 1, 1).Value))
                            missingTitle2 = Trim(CStr(titleVal))
                            If missingTieList = "" Then
                                missingTieList = missingTitle1 & " vs " & missingTitle2
                            Else
                                missingTieList = missingTieList & vbNewLine & missingTitle1 & " vs " & missingTitle2
                            End If
                        End If
                    End If
                End If


ElseIf stmtDiscrepancies = 0 Then
        stmtMsg = vbNewLine & "Discrepancies in Statement to Statement: 0" & _
                  vbNewLine & "Missing tie-throughs in Statement to Statement: " & missingValues \ 2
        If missingValues > 0 Then
            stmtMsgFull = stmtMsg & vbNewLine & vbNewLine & "Missing tie-through detail:" & vbNewLine & missingTieList
        Else
            stmtMsgFull = stmtMsg
        End If
    Else
        stmtMsg = vbNewLine & "Discrepancies in Statement to Statement: " & stmtDiscrepancies & _
                  vbNewLine & "Missing tie-throughs in Statement to Statement: " & missingValues \ 2
        Dim combinedDetail As String
        combinedDetail = ""
        If missingValues > 0 Then
            combinedDetail = combinedDetail & vbNewLine & vbNewLine & "Missing tie-through detail:" & vbNewLine & missingTieList
        End If
        If discrepancyList <> "" Then
            combinedDetail = combinedDetail & vbNewLine & vbNewLine & "Discrepancy detail:" & vbNewLine & discrepancyList
        End If
        stmtMsgFull = stmtMsg & combinedDetail
    End If

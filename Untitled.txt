def find_unreplaced(output_path, prefix="@@"):
    """Find any placeholders still in the output doc that weren't replaced."""
    import re
    unreplaced = set()
    with zipfile.ZipFile(output_path, "r") as z:
        for part in XML_PARTS:
            try:
                data = z.read(part).decode("utf-8")
                matches = re.findall(rf'{re.escape(prefix)}[^@]+{re.escape(prefix)}', data)
                unreplaced.update(matches)
            except:
                pass
    return sorted(unreplaced)

        for ph in unreplaced:

' Read log file and find unreplaced placeholders
    Dim unreplacedMsg As String
    unreplacedMsg = ""
    Dim logFile As Integer
    Dim logLine As String
    Dim unreplacedCount As Long
    unreplacedCount = 0
    
    logFile = FreeFile
    Open logPath For Input As #logFile
    Do While Not EOF(logFile)
        Line Input #logFile, logLine
        If InStr(logLine, "@@") > 0 And InStr(logLine, "UNREPLACED") = 0 Then
            ' Extract placeholder from log line
            Dim startPos As Long
            Dim endPos As Long
            startPos = InStr(logLine, "@@")
            endPos = InStr(startPos + 2, logLine, "@@")
            If startPos > 0 And endPos > 0 Then
                Dim foundPh As String
                foundPh = Mid(logLine, startPos, endPos - startPos + 2)
                Dim tabSource As String
                If phTabMap.Exists(foundPh) Then
                    tabSource = phTabMap(foundPh)
                Else
                    tabSource = "Unknown"
                End If
                unreplacedMsg = unreplacedMsg & vbNewLine & foundPh & " (" & tabSource & ")"
                unreplacedCount = unreplacedCount + 1
            End If
        End If
    Loop
    Close #logFile

    ' Build final message
    Dim skippedMsg As String
    If skippedSheets <> "" Then
        skippedMsg = vbNewLine & vbNewLine & "Skipped tabs (no matching data sheet):" & vbNewLine & skippedSheets
    Else
        skippedMsg = ""
    End If

    Dim unreplacedSummary As String
    If unreplacedCount > 0 Then
        unreplacedSummary = vbNewLine & vbNewLine & "Unreplaced placeholders (" & unreplacedCount & "):" & unreplacedMsg
    Else
        unreplacedSummary = vbNewLine & vbNewLine & "All placeholders replaced successfully."
    End If

        MsgBox "Word document generated successfully!" & vbNewLine & outputPath & skippedMsg & unreplacedSummary, vbInformation, "Done"
    Else
        MsgBox "Script failed. Check log: " & logPath & skippedMsg & unreplacedSummary, vbCritical, "Error"
    End If

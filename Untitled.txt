import openpyxl
import pandas as pd

def import_excel_visible_rows(filepath, sheet_name=0):
    wb = openpyxl.load_workbook(filepath, data_only=True)
    
    # Handle sheet_name as index or string
    if isinstance(sheet_name, int):
        ws = wb.worksheets[sheet_name]
    else:
        ws = wb[sheet_name]
    
    data = []
    for row in ws.iter_rows():
        row_num = row[0].row
        # Skip hidden rows (row_dimensions height == 0 or hidden flag)
        if ws.row_dimensions[row_num].hidden:
            continue
        data.append([cell.value for cell in row])
    
    if not data:
        return pd.DataFrame()
    
    df = pd.DataFrame(data)
    return df

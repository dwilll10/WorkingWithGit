Dim msgContent As String
    Dim endTime As Date
    endTime = Now()

    If exitCode = 0 Then
        msgContent = "Word document generated successfully!" & vbNewLine & outputPath
    Else
        msgContent = "Script failed. Check log: " & logPath
    End If

    msgContent = msgContent & successMsg & partialMsg & skippedMsg & unreplacedSummary

    ' Add timestamps
    Dim fullMsg As String
    fullMsg = "Start time: " & Format(startTime, "yyyy-mm-dd hh:mm:ss") & vbNewLine & _
              "End time:   " & Format(endTime, "yyyy-mm-dd hh:mm:ss") & vbNewLine & vbNewLine & _
              msgContent

    ' Save to txt file
    Dim summaryPath As String
    summaryPath = Environ("USERPROFILE") & "\Documents\automation_summary_" & Format(startTime, "yyyymmdd_hhmmss") & ".txt"
    Dim summaryFile As Integer
    summaryFile = FreeFile
    Open summaryPath For Output As #summaryFile
    Print #summaryFile, fullMsg
    Close #summaryFile

    ' Show MsgBox
    If exitCode = 0 Then
        MsgBox fullMsg, vbInformation, "Done"
    Else
        MsgBox fullMsg, vbCritical, "Error"
    End If

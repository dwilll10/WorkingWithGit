' Remove duplicates from missingTabs
    If missingTabs <> "" Then
        Dim missingArr() As String
        Dim cleanMissing As String
        Dim m As Long
        cleanMissing = ""
        missingArr = Split(missingTabs, vbNewLine)
        For m = 0 To UBound(missingArr)
            Dim mTab As String
            mTab = Trim(missingArr(m))
            If mTab <> "" Then
                If InStr(vbNewLine & cleanMissing & vbNewLine, vbNewLine & mTab & vbNewLine) = 0 Then
                    If cleanMissing = "" Then
                        cleanMissing = mTab
                    Else
                        cleanMissing = cleanMissing & vbNewLine & mTab
                    End If
                End If
            End If
        Next m
        missingTabs = cleanMissing
    End If

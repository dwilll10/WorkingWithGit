Sub RunPythonScript()
    Dim pythonExePath   As String
    Dim pythonScriptPath As String
    Dim workingDirectory As String
    Dim oneDrivePath    As String
    Dim errorLogPath    As String
    Dim command         As String
    Dim wsh             As Object
    Dim result          As Integer
    Dim fileNum         As Integer
    Dim hasErrors       As Boolean
    Dim startTime       As Date
    Dim endTime         As Date

    ' Record start time
    startTime = Now()

    ' Get OneDrive folder path automatically
    oneDrivePath = Environ("OneDrive")

    ' Define paths
    pythonExePath    = """C:\ProgramData\Anaconda3\python.exe"""
    pythonScriptPath = """" & oneDrivePath & "\DHS AFR Automation - Gatekeeper Tool\z. Python Program\GK_new_format\mainScript.py" & """"
    workingDirectory = """" & oneDrivePath & "\DHS AFR Automation - Gatekeeper Tool\z. Python Program\GK_new_format" & """"
    errorLogPath     = oneDrivePath & "\DHS AFR Automation - Gatekeeper Tool\OutputLog_GatekeepingTool.txt"

    ' Pre-clear error log
    fileNum = FreeFile
    Open errorLogPath For Output As #fileNum
    Close fileNum

    ' Build command: run Python and redirect errors to log
    command = "cmd /c cd /d " & workingDirectory & " && " & pythonExePath & " " & pythonScriptPath & " > """ & errorLogPath & """ 2>&1"

    ' Execute command synchronously
    Set wsh = CreateObject("WScript.Shell")
    result = wsh.Run(command, 0, True)

    ' Record end time
    endTime = Now()

    '--------------------------------------------------------------------------
    ' Parse log file
    '--------------------------------------------------------------------------
    Dim logLine         As String
    Dim hasErrors2      As Boolean
    Dim successTabs     As String
    Dim failedTabs      As String
    Dim warningCount    As Long
    Dim errorCount      As Long
    Dim reorderLine     As String

    hasErrors2   = False
    successTabs  = ""
    failedTabs   = ""
    warningCount = 0
    errorCount   = 0
    reorderLine  = ""

    ' Tabs to omit from the popup
    Dim omitTabs As String
    omitTabs = "Start,STMT to STMT,Tie-Through,Tie-Through-fn to fn,Tie-Through fs to fs,Tie-Through Internal"

    If Dir(errorLogPath) <> "" Then
        fileNum = FreeFile
        Open errorLogPath For Input As #fileNum
        Do While Not EOF(fileNum)
            Line Input #fileNum, logLine

            ' Count warnings and errors
            If InStr(logLine, "[WARNING]") > 0 Then warningCount = warningCount + 1
            If InStr(logLine, "[ERROR]") > 0 Then
                errorCount = errorCount + 1
                hasErrors2 = True
            End If

            ' Find failed tabs
            If InStr(logLine, "FAILED:") > 0 And InStr(logLine, "Pipeline") = 0 Then
                Dim failStart As Long
                failStart = InStr(logLine, "FAILED:") + 7
                Dim failName As String
                failName = Trim(Mid(logLine, failStart))
                ' Extract just the tab name (before any colon or space after the name)
                Dim colonPos As Long
                colonPos = InStr(failName, ":")
                If colonPos > 0 Then failName = Trim(Left(failName, colonPos - 1))
                If failedTabs = "" Then
                    failedTabs = failName
                Else
                    failedTabs = failedTabs & vbNewLine & failName
                End If
            End If

            ' Find the reorder complete line
            If InStr(logLine, "Reordering complete. After:") > 0 Then
                reorderLine = logLine
            End If

        Loop
        Close #fileNum
    End If

    '--------------------------------------------------------------------------
    ' Parse the reorder line to get successful tabs
    '--------------------------------------------------------------------------
    If reorderLine <> "" Then
        ' Extract content between [ and ]
        Dim bracketStart As Long
        Dim bracketEnd As Long
        bracketStart = InStr(reorderLine, "[") + 1
        bracketEnd = InStr(reorderLine, "]") - 1
        If bracketStart > 1 And bracketEnd > bracketStart Then
            Dim tabList As String
            tabList = Mid(reorderLine, bracketStart, bracketEnd - bracketStart + 1)

            ' Split by comma and process each tab
            Dim tabArr() As String
            tabArr = Split(tabList, ",")
            Dim t As Long
            For t = 0 To UBound(tabArr)
                Dim tabName As String
                ' Clean up quotes and spaces
                tabName = Trim(tabArr(t))
                tabName = Replace(tabName, "'", "")
                tabName = Trim(tabName)

                ' Skip omitted tabs
                If InStr("," & omitTabs & ",", "," & tabName & ",") = 0 And tabName <> "" Then
                    If successTabs = "" Then
                        successTabs = tabName
                    Else
                        successTabs = successTabs & ", " & tabName
                    End If
                End If
            Next t
        End If
    End If

    '--------------------------------------------------------------------------
    ' Build message
    '--------------------------------------------------------------------------
    Dim statusMsg    As String
    Dim tabsMsg      As String
    Dim failedMsg    As String
    Dim statsMsg     As String
    Dim noteMsg      As String
    Dim fullMsg      As String

    ' Status line
    If result <> 0 Then
        statusMsg = "Script failed with exit code " & result & "." & vbNewLine & _
                    "Check log: " & errorLogPath
    ElseIf hasErrors2 Then
        statusMsg = "Python script completed with errors."
    Else
        statusMsg = "Python script completed successfully."
    End If

    ' Successfully created tabs
    If successTabs <> "" Then
        tabsMsg = vbNewLine & vbNewLine & "Tabs successfully created:" & vbNewLine & successTabs
    Else
        tabsMsg = vbNewLine & vbNewLine & "No completed tabs found in log."
    End If

    ' Failed tabs
    If failedTabs <> "" Then
        failedMsg = vbNewLine & vbNewLine & "Failed tabs:" & vbNewLine & failedTabs
    Else
        failedMsg = ""
    End If

    ' Warning/error stats
    statsMsg = vbNewLine & vbNewLine & _
               "Warnings: " & warningCount & vbNewLine & _
               "Errors:   " & errorCount

    ' End note
    If result <> 0 Then
        noteMsg = vbNewLine & vbNewLine & "NOTE: Automation failed — please check the log file for details."
    ElseIf hasErrors2 Then
        noteMsg = vbNewLine & vbNewLine & "NOTE: Automation completed but errors were encountered. " & _
                  "Please review the failed tabs and log file."
    Else
        noteMsg = vbNewLine & vbNewLine & "NOTE: Automation completed successfully with no errors."
    End If

    ' Assemble full message with timestamps
    fullMsg = "Start time: " & Format(startTime, "yyyy-mm-dd hh:mm:ss") & vbNewLine & _
              "End time:   " & Format(endTime, "yyyy-mm-dd hh:mm:ss") & vbNewLine & _
              "Duration:   " & Format(endTime - startTime, "hh:mm:ss") & vbNewLine & vbNewLine & _
              statusMsg & tabsMsg & failedMsg & statsMsg & noteMsg & vbNewLine & vbNewLine & _
              "Log file: " & errorLogPath

    '--------------------------------------------------------------------------
    ' Save summary to txt file
    '--------------------------------------------------------------------------
    Dim summaryPath As String
    summaryPath = oneDrivePath & "\DHS AFR Automation - Gatekeeper Tool\GK_summary_" & _
                  Format(startTime, "yyyymmdd_hhmmss") & ".txt"
    Dim summaryFile As Integer
    summaryFile = FreeFile
    Open summaryPath For Output As #summaryFile
    Print #summaryFile, fullMsg
    Close #summaryFile

    fullMsg = fullMsg & vbNewLine & "Summary file: " & summaryPath

    '--------------------------------------------------------------------------
    ' Show MsgBox
    '--------------------------------------------------------------------------
    If result <> 0 Or hasErrors2 Then
        MsgBox fullMsg, vbExclamation, "Script Completed with Errors"
    Else
        MsgBox fullMsg, vbInformation, "Execution Completed"
    End If

    Set wsh = Nothing
End Sub

def reorder_sheets(wb, desired_order):
    existing = [s.name for s in wb.sheets]
    order = [name for name in desired_order if name in existing]
    
    if not order:
        logger.info("No sheets from desired_order found in workbook.")
        return
    
    logger.info(f"Reordering. Before: {[s.name for s in wb.sheets]}")
    
    # Grab a stable COM reference to each sheet BEFORE any moves happen.
    # Using the Sheets collection via the workbook's .api avoids xlwings re-lookups.
    sheet_refs = {s.name: s.api for s in wb.sheets}
    
    # Strategy: move each sheet in reverse order, placing it BEFORE the first sheet.
    # After the loop, they end up in the correct forward order at the front of the workbook.
    # Sheets not in `order` drift to the end naturally.
    try:
        for name in reversed(order):
            sheet_api = sheet_refs[name]
            # Re-fetch the first sheet each time — it changes as we move things
            first_sheet_api = wb.sheets[0].api
            if sheet_api is not first_sheet_api:
                sheet_api.Move(Before=first_sheet_api)
    except Exception as e:
        logger.error(f"Failed during reorder: {e}")
        import traceback
        logger.error(traceback.format_exc())
        return
    
    logger.info(f"Reordering complete. After: {[s.name for s in wb.sheets]}")

def formatResult(sheet, dataframe):
    try:
        # Check if the sheet is None
        if sheet is None:
            logger.error("Error: sheet is None. Cannot format headers.")
            return

        # Get the number of columns
        num_columns = sheet.cells(1, 1).end('right').column  # Assuming headers are in the first row
        
        # Bold the headers
        for col in range(1, num_columns + 1):
            sheet.cells(1, col).font.bold = True

        # Set font size for headers
        for col in range(1, num_columns + 1):
            sheet.cells(1, col).font.size = 12

        # Focus on column 2 for debugging purposes
        col_to_adjust = 2
        
     

        # Check if the dataframe has enough columns
        if dataframe.shape[1] >= col_to_adjust:
            # Get the max length of the values in the dataframe for column 2
            max_length = max(dataframe.iloc[:, col_to_adjust - 1].astype(str).map(len).max(), len(sheet.cells(1, col_to_adjust).value))
                  
            
        else:
            max_length = len(sheet.cells(1, col_to_adjust).value)
  

        # Apply the calculated width with some padding (e.g., 4)
        sheet.range(sheet.cells(1, col_to_adjust), sheet.cells(1, col_to_adjust)).column_width = max_length + 4
  
        
        # Initialize variables for columns
        bs_amount_col = None
        fn_amount_col = None
        fn_amount_one_col = None
        fn_amount_two_col = None
        delta_col = None
        fs_amount_one_col = None  # New column
        fs_amount_two_col = None  # New column
        rn_amount_one_col = None  # New column for RN Amount One
        rn_amount_two_col = None  # New column for RN Amount Two

        # Check the headers and identify the relevant columns
        for col in range(1, num_columns + 1):
            header_value = sheet.cells(1, col).value
            if header_value == 'FS Amount':
                bs_amount_col = col
            elif header_value == 'FN Amount':
                fn_amount_col = col
            elif header_value == 'FN Amount One':
                fn_amount_one_col = col
            elif header_value == 'FN Amount Two':
                fn_amount_two_col = col
            elif header_value == 'Difference':
                delta_col = col
            elif header_value == 'FS Amount One':  # New header check
                fs_amount_one_col = col
            elif header_value == 'FS Amount Two':  # New header check
                fs_amount_two_col = col
            elif header_value == 'RN Amount One':  # New header check for RN Amount One
                rn_amount_one_col = col
            elif header_value == 'RN Amount Two':  # New header check for RN Amount Two
                rn_amount_two_col = col

        # List of columns that must exist at least one
        required_columns_found = any([bs_amount_col, fn_amount_one_col, fs_amount_one_col, rn_amount_one_col])

        # Check if we found any of the required columns
        if not required_columns_found or delta_col is None:
            logger.error("Error: One or more required columns not found.")
            return

        # Apply accounting format to columns if found
        last_row = sheet.cells(1, 1).end('down').row

        # Accounting number format
        accounting_format = '_-[$$-409]* #,##0.00_-;-[$$-409]* #,##0.00_-;_-[$$-409]* "-"_-;_-@_-'

        # Apply accounting format to available columns
        if bs_amount_col is not None:
            sheet.range(sheet.cells(2, bs_amount_col), sheet.cells(last_row, bs_amount_col)).number_format = accounting_format
        if fn_amount_col is not None:
            sheet.range(sheet.cells(2, fn_amount_col), sheet.cells(last_row, fn_amount_col)).number_format = accounting_format
        if fn_amount_one_col is not None:
            sheet.range(sheet.cells(2, fn_amount_one_col), sheet.cells(last_row, fn_amount_one_col)).number_format = accounting_format
        if fn_amount_two_col is not None:
            sheet.range(sheet.cells(2, fn_amount_two_col), sheet.cells(last_row, fn_amount_two_col)).number_format = accounting_format
        if delta_col is not None:
            sheet.range(sheet.cells(2, delta_col), sheet.cells(last_row, delta_col)).number_format = accounting_format
        if fs_amount_one_col is not None:  # Apply accounting format to FS Amount One
            sheet.range(sheet.cells(2, fs_amount_one_col), sheet.cells(last_row, fs_amount_one_col)).number_format = accounting_format
        if fs_amount_two_col is not None:  # Apply accounting format to FS Amount Two
            sheet.range(sheet.cells(2, fs_amount_two_col), sheet.cells(last_row, fs_amount_two_col)).number_format = accounting_format
        if rn_amount_one_col is not None:  # Apply accounting format to RN Amount One
            sheet.range(sheet.cells(2, rn_amount_one_col), sheet.cells(last_row, rn_amount_one_col)).number_format = accounting_format
        if rn_amount_two_col is not None:  # Apply accounting format to RN Amount Two
            sheet.range(sheet.cells(2, rn_amount_two_col), sheet.cells(last_row, rn_amount_two_col)).number_format = accounting_format

        # Adjust column widths for formatted columns
        for col in [bs_amount_col, fn_amount_col, fn_amount_one_col, fn_amount_two_col, delta_col, fs_amount_one_col, fs_amount_two_col, rn_amount_one_col, rn_amount_two_col]:
            if col is not None:  # Check that col is valid
                try:
                    # Adjust column width based on max length
                    max_length = max(dataframe.iloc[:, col - 1].astype(str).map(len).max(), len(sheet.cells(1, col).value))  # Including header
                except IndexError:
                    # Handle if column index exceeds dataframe columns
                    max_length = len(sheet.cells(1, col).value)

                # Add a little padding to the column width
                sheet.range(sheet.cells(1, col), sheet.cells(1, col)).column_width = max_length + 4

        # Calculate totals and write them at the bottom
        total_row = last_row + 2
        if bs_amount_col is not None:
            sheet.cells(total_row, bs_amount_col).formula = f'=SUM({sheet.cells(2, bs_amount_col).address}:{sheet.cells(last_row, bs_amount_col).address})'
        if fn_amount_col is not None:
            sheet.cells(total_row, fn_amount_col).formula = f'=SUM({sheet.cells(2, fn_amount_col).address}:{sheet.cells(last_row, fn_amount_col).address})'
        if fn_amount_one_col is not None:
            sheet.cells(total_row, fn_amount_one_col).formula = f'=SUM({sheet.cells(2, fn_amount_one_col).address}:{sheet.cells(last_row, fn_amount_one_col).address})'
        if fn_amount_two_col is not None:
            sheet.cells(total_row, fn_amount_two_col).formula = f'=SUM({sheet.cells(2, fn_amount_two_col).address}:{sheet.cells(last_row, fn_amount_two_col).address})'
        if delta_col is not None:
            sheet.cells(total_row, delta_col).formula = f'=SUM({sheet.cells(2, delta_col).address}:{sheet.cells(last_row, delta_col).address})'
        if fs_amount_one_col is not None:  # Calculate total for FS Amount One
            sheet.cells(total_row, fs_amount_one_col).formula = f'=SUM({sheet.cells(2, fs_amount_one_col).address}:{sheet.cells(last_row, fs_amount_one_col).address})'
        if fs_amount_two_col is not None:  # Calculate total for FS Amount Two
            sheet.cells(total_row, fs_amount_two_col).formula = f'=SUM({sheet.cells(2, fs_amount_two_col).address}:{sheet.cells(last_row, fs_amount_two_col).address})'
        if rn_amount_one_col is not None:  # Calculate total for RN Amount One
            sheet.cells(total_row, rn_amount_one_col).formula = f'=SUM({sheet.cells(2, rn_amount_one_col).address}:{sheet.cells(last_row, rn_amount_one_col).address})'
        if rn_amount_two_col is not None:  # Calculate total for RN Amount Two
            sheet.cells(total_row, rn_amount_two_col).formula = f'=SUM({sheet.cells(2, rn_amount_two_col).address}:{sheet.cells(last_row, rn_amount_two_col).address})'

        # Bold the totals row
        for col in [bs_amount_col, fn_amount_col, fn_amount_one_col, fn_amount_two_col, delta_col, fs_amount_one_col, fs_amount_two_col, rn_amount_one_col, rn_amount_two_col]:
            if col is not None:  # Check that col is valid
                sheet.cells(total_row, col).font.bold = True

    except Exception as e:
        logger.error(f"Error in formatResult: {e}")

def updateFile(active_sheet_name='Start'):

        ]

            raise ValueError(f"No open workbook contains a sheet named '{active_sheet_name}'.")
        elif len(matching_books) > 1:
            names = [wb.name for wb in matching_books]
            raise ValueError(
                f"Multiple open workbooks contain a sheet named '{active_sheet_name}': {names}. "
                f"Please close the others and try again."

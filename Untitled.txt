Function GetSum(wsData As Worksheet, sections As Variant, subtitles As Variant) As Double
    Dim total As Double
    Dim i As Long
    Dim foundCell As Range
    Dim nextCell As Range
    Dim sectionRow As Long
    Dim nextSectionRow As Long
    Dim subtitleRow As Long
    Dim j As Long
    
    total = 0
    For i = 0 To UBound(sections)
        Set foundCell = wsData.Columns("A").Find( _
            What:=sections(i), LookIn:=xlValues, LookAt:=xlWhole, MatchCase:=True)
        If foundCell Is Nothing Then GoTo NextItem
        
        sectionRow = foundCell.Row
        Set nextCell = wsData.Columns("A").FindNext(foundCell)
        If nextCell Is Nothing Or nextCell.Row <= sectionRow Then
            nextSectionRow = wsData.Cells(wsData.Rows.Count, "C").End(xlUp).Row
        Else
            nextSectionRow = nextCell.Row - 1
        End If
        
        subtitleRow = 0
        For j = sectionRow To nextSectionRow
            If Trim(wsData.Cells(j, 3).Value) = subtitles(i) Then
                subtitleRow = j
                Exit For
            End If
        Next j
        
        If subtitleRow = 0 Then GoTo NextItem
        total = total + wsData.Cells(subtitleRow, 4).Value
NextItem:
    Next i
    GetSum = total
End Function

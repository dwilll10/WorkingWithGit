def find_unreplaced(output_path, prefix="@@"):
    """Find any placeholders still in the output doc that weren't replaced."""
    import re
    unreplaced = set()
    with zipfile.ZipFile(output_path, "r") as z:
        for part in XML_PARTS:
            try:
                data = z.read(part).decode("utf-8")
                
                # First check raw XML for clean placeholders
                clean_matches = re.findall(rf'{re.escape(prefix)}\w+{re.escape(prefix)}', data)
                unreplaced.update(clean_matches)
                
                # Also extract just text content to catch split-run placeholders
                from lxml import etree
                W = "http://schemas.openxmlformats.org/wordprocessingml/2006/main"
                tree = etree.fromstring(data.encode("utf-8"))
                
                # Check each paragraph's full text
                for para in tree.iter(f"{{{W}}}p"):
                    runs = para.findall(f".//{{{W}}}r")
                    text_runs = [r for r in runs if r.find(f"{{{W}}}t") is not None]
                    full_text = "".join((r.find(f"{{{W}}}t").text or "") for r in text_runs)
                    split_matches = re.findall(rf'{re.escape(prefix)}\w+{re.escape(prefix)}', full_text)
                    unreplaced.update(split_matches)
                    
            except Exception as e:
                print(f"  Warning: could not check {part}: {e}")
                
    return sorted(unreplaced)

    Open logPath For Input As #logFile

        Line Input #logFile, logLine
        ' Detect start of unreplaced section
        If InStr(logLine, "UNREPLACED PLACEHOLDERS") > 0 Then
            inUnreplacedSection = True
        ' Detect end of unreplaced section
        ElseIf InStr(logLine, "End VBA Log") > 0 Or InStr(logLine, "All placeholders were replaced") > 0 Then
            inUnreplacedSection = False
        ' Only collect lines within the unreplaced section
        ElseIf inUnreplacedSection And Left(Trim(logLine), 2) = "@@" Then
            foundPh = Trim(logLine)
            If phTabMap.Exists(foundPh) Then
                tabSource = phTabMap(foundPh)
            Else
                tabSource = "Unknown"
            End If
            unreplacedMsg = unreplacedMsg & vbNewLine & foundPh & " (" & tabSource & ")"
            unreplacedCount = unreplacedCount + 1
        End If
    Loop

For Each ws In wbConfig.Worksheets

        sheetName = ws.Name

        ' Skip sheets handled by separate macros or the Sums block
        If sheetName = "OI-SOS" Then GoTo NextSheet
        If sheetName = "Sums" Then GoTo NextSheet
        WriteLog "Start processing sheet '" & sheetName & "'"

        ' Check if this config tab exists in the master list
        dataExists = False
        For m = 2 To masterLastRow
            If LCase(Trim(wsMaster.Cells(m, masterTitleCol).Value)) = LCase(Trim(sheetName)) Then
                dataExists = True
                Exit For
            End If
        Next m

        ' Map placeholders to config tab regardless of whether data sheet exists
        Dim mapWsConfig As Worksheet
        Set mapWsConfig = ws
        Dim mapLastRow As Long
        mapLastRow = mapWsConfig.Cells(mapWsConfig.Rows.Count, "C").End(xlUp).Row
        Dim mapRow As Long
        For mapRow = 2 To mapLastRow
            Dim mapPh As Long
            For mapPh = 3 To 7
                Dim mapPhVal As String
                mapPhVal = Trim(CStr(mapWsConfig.Cells(mapRow, mapPh).Value))
                If mapPhVal <> "" Then
                    If Not phTabMap.Exists(mapPhVal) Then
                        phTabMap.Add mapPhVal, sheetName
                    End If
                End If
            Next mapPh
            ' Also check K+ columns
            Dim mapExtra As Long
            mapExtra = 11
            Do While Trim(CStr(mapWsConfig.Cells(mapRow, mapExtra).Value)) <> ""
                mapPhVal = Trim(CStr(mapWsConfig.Cells(mapRow, mapExtra).Value))
                If Not phTabMap.Exists(mapPhVal) Then
                    phTabMap.Add mapPhVal, sheetName
                End If
                mapExtra = mapExtra + 1
            Loop
        Next mapRow

        If Not dataExists Then
            WriteLog "SKIPPED: Config tab '" & sheetName & "' has placeholders but there is no matching entry in BSFNMasterList."
            If skippedSheets = "" Then
                skippedSheets = sheetName
            Else
                skippedSheets = skippedSheets & ", " & sheetName
            End If
            GoTo NextSheet
        End If

        ' Check that ThisWorkbook actually has a matching sheet
        Dim wsDataExists As Boolean
        wsDataExists = False
        Dim checkWs As Worksheet
        For Each checkWs In ThisWorkbook.Worksheets
            If LCase(checkWs.Name) = LCase(sheetName) Then
                wsDataExists = True
                Exit For
            End If
        Next checkWs

        If Not wsDataExists Then
            WriteLog "SKIPPED: Config tab '" & sheetName & "' found in master list but no matching sheet in ThisWorkbook."
            If skippedSheets = "" Then
                skippedSheets = sheetName
            Else
                skippedSheets = skippedSheets & ", " & sheetName
            End If
            GoTo NextSheet
        End If

        Set wsConfig = ws
        Set wsData = ThisWorkbook.Sheets(sheetName)

import sys
import zipfile
import os
from lxml import etree

W = "http://schemas.openxmlformats.org/wordprocessingml/2006/main"

word_path   = sys.argv[1]
output_path = sys.argv[2]

if os.path.exists(output_path):
    os.remove(output_path)

# Only process document.xml — parse and reserialize it with NO replacements
# Everything else copied raw
with zipfile.ZipFile(word_path, "r") as zin:
    with zipfile.ZipFile(output_path, "w", zipfile.ZIP_DEFLATED) as zout:
        for item in zin.infolist():
            data = zin.read(item.filename)
            if item.filename == "word/document.xml":
                print(f"Parsing {item.filename}...")
                tree = etree.fromstring(data)
                data = etree.tostring(tree, encoding="unicode").encode("utf-8")
                print(f"Done. New size: {len(data)}")
            zout.writestr(item, data)

print(f"Written to {output_path}")

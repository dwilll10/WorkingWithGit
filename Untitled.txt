def get_pending_fn_filters(existing_sheets: set[str], wb) -> list[str]:
    """
    Returns the list of fn_filter keys whose output sheet is NOT yet in the workbook.
    Uses the same Config mapping that import_fn/import_fs already reads.
    """
    mapping_file = dir_p.mapping_table()
    df_fn  = pd.read_excel(mapping_file, sheet_name='Config')
    df_fs  = pd.read_excel(mapping_file, sheet_name='Config-FS')

    fn_keys = df_fn['report_key'].str.strip().str.lower().tolist()
    fs_keys = df_fs['report_key'].str.strip().str.lower().tolist()

    def to_sheet_name(key: str) -> str:
        return key.upper()

    pending_fn = [k for k in fn_keys if to_sheet_name(k) not in existing_sheets]
    pending_fs = [k for k in fs_keys if to_sheet_name(k) not in existing_sheets]

    return pending_fn, pending_fs


def df_to_sheet(dataframe, sheet_name, wb, after_sheet_name='Start', skip_if_exists=True):
    try:
        dataframe = dataframe.reset_index(drop=True)

        existing_sheets = [sheet.name for sheet in wb.sheets]
        if sheet_name in existing_sheets:
            if skip_if_exists:
                logger.info(f"Sheet '{sheet_name}' already exists. Skipping")
                return None
            wb.sheets[sheet_name].delete()
            time.sleep(0.1)

        if after_sheet_name in existing_sheets:
            after_index = existing_sheets.index(after_sheet_name)
            new_sheet = wb.sheets.add(sheet_name, after=wb.sheets[after_index])
        else:
            new_sheet = wb.sheets.add(sheet_name)

        new_sheet.range('A1').options(index=False, header=True).value = dataframe

        for i in range(dataframe.shape[1]):
            column_values = dataframe.iloc[:, i].fillna('').astype(str)
            max_length = max(column_values.map(len).max(), len(str(dataframe.columns[i])))
            new_sheet.range(new_sheet.cells(1, i + 1), new_sheet.cells(1, i + 1)).column_width = max_length + 2

        if sheet_name not in ['FN10', 'FN19', 'FN21', 'FN25', 'FN30SNCP', 'RSI-SCA', 'RSI-LAND', 'FN30SNC']:
            new_sheet.range('B:B').column_width = 5
            new_sheet.range(f'B1:B{dataframe.shape[0] + 1}').api.HorizontalAlignment = xw.constants.HAlign.xlHAlignCenter

        fc.cur(new_sheet, dataframe)

        header_range = f'A1:{new_sheet.cells(1, dataframe.shape[1]).address}'
        new_sheet.range(header_range).api.Font.Bold = True

        if sheet_name == "STMT to STMT":
            new_sheet.range(header_range).api.Interior.Color = 16763904
            new_sheet.range('1:1').row_height = 45
            new_sheet.range('A:A').column_width = 63
            new_sheet.range('B:B').column_width = 18
            new_sheet.range('C:C').column_width = 17
            new_sheet.range('D:D').column_width = 39
            new_sheet.range('A:A').api.WrapText = True

            new_sheet.activate()
            new_sheet.api.Application.ActiveWindow.FreezePanes = False
            new_sheet.range('A2').select()
            new_sheet.api.Application.ActiveWindow.FreezePanes = True

            header_range_obj = new_sheet.range(header_range)
            header_range_obj.api.HorizontalAlignment = xw.constants.HAlign.xlHAlignCenter
            header_range_obj.api.VerticalAlignment = xw.constants.VAlign.xlVAlignCenter

            new_sheet.range(header_range).api.Borders(xw.constants.BordersIndex.xlEdgeBottom).LineStyle = 1
            new_sheet.range(header_range).api.Borders(xw.constants.BordersIndex.xlEdgeBottom).Color = 0
            new_sheet.range(header_range).api.Borders(xw.constants.BordersIndex.xlEdgeBottom).Weight = 2

            for col in range(1, dataframe.shape[1]):
                col_range = new_sheet.range(new_sheet.cells(1, col), new_sheet.cells(1, col + 1))
                col_range.api.Borders(xw.constants.BordersIndex.xlEdgeRight).LineStyle = 1
                col_range.api.Borders(xw.constants.BordersIndex.xlEdgeRight).Color = 0
                col_range.api.Borders(xw.constants.BordersIndex.xlEdgeRight).Weight = 2

            new_sheet.range('A1').api.Borders(xw.constants.BordersIndex.xlEdgeRight).LineStyle = 1
            new_sheet.range('A1').api.Borders(xw.constants.BordersIndex.xlEdgeRight).Color = 0
            new_sheet.range('A1').api.Borders(xw.constants.BordersIndex.xlEdgeRight).Weight = 2

            difference_column_index = 3
            last_data_row = dataframe.shape[0] + 1
            difference_range = new_sheet.range(
                new_sheet.cells(2, difference_column_index),
                new_sheet.cells(last_data_row, difference_column_index)
            )
            difference_values = difference_range.value
            if not isinstance(difference_values, list):
                difference_values = [difference_values]

            col_a_range = new_sheet.range(
                new_sheet.cells(2, 1),
                new_sheet.cells(last_data_row + 5, 1)
            )
            col_a_values = col_a_range.value
            if not isinstance(col_a_values, list):
                col_a_values = [col_a_values]

            valid_values = ["Tie-Through FS to FN", "Tie-Through FN to FN",
                            "Tie-Through FS to FS", "Internal Tie-Through"]

            border_plans = []
            delete_rows = []

            for i, difference_value in enumerate(difference_values):
                row = i + 2
                if difference_value in [None, '']:
                    continue
                lookahead_idx = i + 3
                value_3_down = col_a_values[lookahead_idx] if lookahead_idx < len(col_a_values) else None
                if value_3_down in valid_values:
                    border_plans.append({
                        'row': row,
                        'move_value_to_row': row + 2,
                        'move_value': value_3_down,
                        'border_end_col': 4,
                    })
                    delete_rows.append(row + 3)
                else:
                    border_plans.append({
                        'row': row,
                        'move_value_to_row': None,
                        'move_value': None,
                        'border_end_col': 3,
                    })

            for plan in border_plans:
                row = plan['row']
                col3_cell = new_sheet.cells(row, difference_column_index)

                bottom_border = col3_cell.api.Borders(xw.constants.BordersIndex.xlEdgeBottom)
                bottom_border.LineStyle = xw.constants.LineStyle.xlDouble
                bottom_border.Color = 0
                bottom_border.Weight = 4

                top_border = col3_cell.api.Borders(xw.constants.BordersIndex.xlEdgeTop)
                top_border.LineStyle = 1
                top_border.Color = 0
                top_border.Weight = 2

                new_sheet.cells(row, difference_column_index + 1).api.Interior.Color = 13434828

                if plan['move_value'] is not None:
                    new_sheet.cells(plan['move_value_to_row'], 1).value = plan['move_value']
                    new_sheet.cells(plan['move_value_to_row'], 1).api.Font.Bold = True

                next_row_range = new_sheet.range(
                    new_sheet.cells(row + 1, 1),
                    new_sheet.cells(row + 1, plan['border_end_col'])
                )
                next_row_border = next_row_range.api.Borders(xw.constants.BordersIndex.xlEdgeBottom)
                next_row_border.LineStyle = 1
                next_row_border.Color = 0
                next_row_border.Weight = 4 if plan['border_end_col'] == 4 else 3

            for row_to_delete in sorted(delete_rows, reverse=True):
                new_sheet.range(f"A{row_to_delete}").api.EntireRow.Delete()

            value_in_row_3 = new_sheet.cells(3, 1).value
            if value_in_row_3 not in [None, '']:
                new_sheet.cells(2, 1).value = value_in_row_3
                new_sheet.cells(2, 1).api.Font.Bold = True
                new_sheet.range("A3").api.EntireRow.Delete()

        if 'Start' in existing_sheets:
            wb.sheets['Start'].activate()

        return new_sheet

    except Exception as e:
        logger.error(f"Error in df_to_sheet: {e}")
        return None


def reorder_sheets(wb, desired_order):
    existing = [s.name for s in wb.sheets]
    order = [name for name in desired_order if name in existing]

    if not order:
        logger.info("No sheets from desired_order found in workbook.")
        return

    logger.info(f"Reordering. Before: {[s.name for s in wb.sheets]}")

    sheet_refs = {s.name: s.api for s in wb.sheets}

    try:
        for name in reversed(order):
            sheet_api = sheet_refs[name]
            first_sheet_api = wb.sheets[0].api
            if sheet_api is not first_sheet_api:
                sheet_api.Move(Before=first_sheet_api)
    except Exception as e:
        logger.error(f"Failed during reorder: {e}")
        import traceback
        logger.error(traceback.format_exc())
        return

    logger.info(f"Reordering complete. After: {[s.name for s in wb.sheets]}")

                lambda: RR_updateFn.fn11('FN11', wb)                 if 'fn11'         in pending_fn else None,
                lambda: RR_updateFn.fn10('FN10', wb)                 if 'fn10'         in pending_fn else None,
                lambda: RR_updateFn.fn9('FN9', wb)                   if 'fn9'          in pending_fn else None,
                lambda: RR_updateFn.fn8('FN8', wb)                   if 'fn8'          in pending_fn else None,
                lambda: RR_updateFn.fn7('FN7', wb)                   if 'fn7'          in pending_fn else None,
                lambda: RR_updateFn.fn6('FN6', wb)                   if 'fn6'          in pending_fn else None,
                lambda: RR_updateFn.fn5('FN5', wb)                   if 'fn5'          in pending_fn else None,
                lambda: RR_updateFn.fn4('FN4', wb)                   if 'fn4'          in pending_fn else None,
                lambda: RR_updateFn.fn3('FN3', wb)                   if 'fn3'          in pending_fn else None,
                lambda: RR_updateFn.fn2('FN2', wb)                   if 'fn2'          in pending_fn else None,
                lambda: RR_updateFs.scaFormat('SCA', wb)             if 'sca'          in pending_fs else None,
                lambda: RR_updateFs.sbrFormat('SBR', wb)             if 'sbr'          in pending_fs else None,
                lambda: RR_updateFs.scnpFormat('SCNP', wb)           if 'scnp'         in pending_fs else None,
                lambda: RR_updateFs.sncFormat('SNC', wb)             if 'snc'          in pending_fs else None,
                lambda: RR_updateFs.bsFormat('BS', wb)               if 'bs'           in pending_fs else None,
            ]

            t0 = time.time()
            tracker = ru.PipelineTracker()
            fn_names = [
                'RSI-DEFERRED', 'RSI-SBR', 'OI-SOS', 'RSI-LAND', 'RSI-SCA',
                'FN32', 'FN30SCNP', 'FN30SNC', 'FN29', 'FN28', 'FN27', 'FN26',
                'FN25', 'FN23', 'FN22', 'FN21', 'FN20', 'FN19', 'FN18', 'FN17',
                'FN16', 'FN15', 'FN14', 'FN13', 'FN12', 'FN11', 'FN10', 'FN9',
                'FN8', 'FN7', 'FN6', 'FN5', 'FN4', 'FN3', 'FN2',
                'SCA', 'SBR', 'SCNP', 'SNC', 'BS'
            ]
            for name, fn in zip(fn_names, function_calls):
                t_fn = time.time()
                result = tracker.run(name, fn)
                if result is not None:
                    sheets_written_this_run.add(name)
                logger.info(f"TIMING: {name} = {time.time() - t_fn:.2f}s")
            tracker.summary()
            logger.info(f"TIMING: full pipeline = {time.time() - t0:.2f}s")
            logger.info(f"Sheets written this run: {sheets_written_this_run}")

            # ----------------------------------------------------------------
            # Tie-through sheets — recompute if any source sheet was written,
            # or create fresh if the sheet doesn't exist yet.
            # ----------------------------------------------------------------
            tabName = 'Tie-Through'
            if tabName not in existing_sheets or should_recompute_tie(tabName):
                if tabName in existing_sheets:
                    wb.sheets[tabName].delete()
                    logger.info(f"Deleted '{tabName}' for recompute — source sheets changed.")
                t0 = time.time()
                fsTofn = Tie_bs_fn.tie_through(tabName, wb)
                logger.info(f"TIMING: Tie_bs_fn.tie_through = {time.time() - t0:.2f}s")
                t0 = time.time()
                write_tie_sheet(tabName, fsTofn)
                logger.info(f"TIMING: Write tie through sheet = {time.time() - t0:.2f}s")
            else:
                logger.info(f"Skipping '{tabName}' — sheet exists and no source sheets changed.")
                fsTofn = pd.DataFrame()

            tabName = 'Tie-Through-fn to fn'
            if tabName not in existing_sheets or should_recompute_tie(tabName):
                if tabName in existing_sheets:
                    wb.sheets[tabName].delete()
                    logger.info(f"Deleted '{tabName}' for recompute — source sheets changed.")
                t0 = time.time()
                fnTofn = Tie_fn_fn.tie_through(tabName, wb)
                logger.info(f"TIMING: Tie_fn_fn.tie_through = {time.time() - t0:.2f}s")
                t0 = time.time()
                write_tie_sheet(tabName, fnTofn)
                logger.info(f"TIMING: Write fn-fn tie through = {time.time() - t0:.2f}s")
            else:
                logger.info(f"Skipping '{tabName}' — sheet exists and no source sheets changed.")
                fnTofn = pd.DataFrame()

            tabName = 'Tie-Through fs to fs'
            if tabName not in existing_sheets or should_recompute_tie(tabName):
                if tabName in existing_sheets:
                    wb.sheets[tabName].delete()
                    logger.info(f"Deleted '{tabName}' for recompute — source sheets changed.")
                t0 = time.time()
                fsTofs = Tie_fs_fs.tie_through(tabName, wb)
                logger.info(f"TIMING: Tie_fs_fs.tie_through = {time.time() - t0:.2f}s")
                t0 = time.time()
                write_tie_sheet(tabName, fsTofs)
                logger.info(f"TIMING: Write fs-fs tie through = {time.time() - t0:.2f}s")
            else:
                logger.info(f"Skipping '{tabName}' — sheet exists and no source sheets changed.")
                fsTofs = pd.DataFrame()

            tabName = 'Tie-Through Internal'
            if tabName not in existing_sheets or should_recompute_tie(tabName):
                if tabName in existing_sheets:
                    wb.sheets[tabName].delete()
                    logger.info(f"Deleted '{tabName}' for recompute — source sheets changed.")
                t0 = time.time()
                internal_data = internal_fn.tie_through(tabName, wb)
                logger.info(f"TIMING: Tie_internal.tie_through = {time.time() - t0:.2f}s")
                t0 = time.time()
                write_tie_sheet(tabName, internal_data)
                logger.info(f"TIMING: Write internal tie through = {time.time() - t0:.2f}s")
            else:
                logger.info(f"Skipping '{tabName}' — sheet exists and no source sheets changed.")
                internal_data = pd.DataFrame()

            # Debug: check for FN11 references across tie-through DataFrames
            for df_name, df in [('fsTofn', fsTofn), ('fnTofn', fnTofn), ('fsTofs', fsTofs), ('internal_data', internal_data)]:
                for col in df.columns:
                    mask = df[col].astype(str).str.contains('FN11', na=False, case=False)
                    if mask.any():
                        logger.info(f"  FN11 found in {df_name}.{col}: {mask.sum()} rows")

            # ----------------------------------------------------------------
            # STMT to STMT — recompute if any tie-through was recomputed,
            # or create fresh if the sheet doesn't exist yet.
            # ----------------------------------------------------------------
            tie_throughs_changed = any(should_recompute_tie(t) or t not in existing_sheets
                                       for t in tie_through_sources)

            tabName = 'STMT to STMT'
            if tabName in existing_sheets and not tie_throughs_changed:
                logger.info(f"Skipping '{tabName}' — sheet exists and no tie-throughs changed.")
            elif any(df.empty for df in [fsTofn, fnTofn, fsTofs, internal_data]):
                logger.warning(f"Skipping '{tabName}' — one or more tie-through DataFrames are empty.")
            else:
                if tabName in existing_sheets:
                    wb.sheets[tabName].delete()
                    logger.info(f"Deleted '{tabName}' for recompute — tie-through sheets changed.")
                t0 = time.time()
                fsFn = stmt.fs_to_fn(fsTofn)
                fnFn = stmt.fn_to_fn(fnTofn)
                fsFs = stmt.fs_to_fs(fsTofs)
                inte = stmt.internal_tie(internal_data)
                main = stmt.stmt_tie(fsFn, fnFn, fsFs, inte)
                logger.info(f"main: {len(main)} rows")
                sheet = df_to_sheet(main, tabName, wb, skip_if_exists=False)
                logger.info(f"TIMING: STMT TO STMT = {time.time() - t0:.2f}s")

            reorder_sheets(wb, desired_order)
            logger.info(f"TIMING: overall = {time.time() - overall_start:.2f}s")
            wb.save()

        finally:
            app.screen_updating = True
            app.calculation = 'automatic'
            app.display_alerts = True
            app.api.EnableEvents = True

    except Exception as e:
        logger.error(f"Error in updateFile: {e}")

Sub ExportOI_SOS()
    Dim pythonPath As String
    Dim scriptPath As String
    Dim logPath As String
    Dim configPath As String
    Dim command As String
    Dim wsh As Object
    Dim oneDrivePath As String
    Dim wsData As Worksheet
    Dim wsConfig As Worksheet
    Dim wbConfig As Workbook
    Dim exitCode As Long
    Dim args As String
    Dim placeholder As String
    Dim rowNum As Integer
    Dim colOffset As Integer
    Dim value As String
    Dim lastRow As Long
    Dim i As Long

    oneDrivePath = Environ("OneDrive")
    pythonPath = "C:\ProgramData\Anaconda3\python.exe"
    scriptPath = oneDrivePath & "\Documents\z. Python Program\GK_new_format\WordExportOISOS.py"
    logPath = Environ("USERPROFILE") & "\Documents\error_log_wordexport.txt"
    configPath = oneDrivePath & "\Documents\z. Python Program\GK_new_format\config.xlsx"  ' <-- adjust path as needed

    ' Reference data sheet in the already open workbook
    Set wsData = ThisWorkbook.Sheets("OI-SOS")

    ' Open config workbook silently, read, then close
    Application.ScreenUpdating = False

    On Error GoTo ConfigError
    Set wbConfig = Workbooks.Open(configPath, ReadOnly:=True)
    On Error GoTo 0

    Set wsConfig = wbConfig.Sheets(1)  ' <-- adjust sheet index/name as needed

    ' Read col_offset from a fixed cell, e.g. E1
    colOffset = CInt(wsConfig.Range("E1").Value)  ' <-- adjust cell as needed

    ' Find last row of config table (columns A and B)
    lastRow = wsConfig.Cells(wsConfig.Rows.Count, "A").End(xlUp).Row

    ' Build args by iterating config rows (skip header in row 1)
    args = ""
    For i = 2 To lastRow
        placeholder = Trim(wsConfig.Cells(i, 1).Value)   ' Column A: placeholder
        rowNum = CInt(wsConfig.Cells(i, 2).Value)         ' Column B: row number

        If placeholder = "" Then GoTo NextRow

        ' Read value from OI-SOS sheet: column C (3) + offset
        value = wsData.Cells(rowNum, 3 + colOffset).Value

        ' Append as placeholder=value pair
        args = args & " """ & placeholder & "=" & value & """"

NextRow:
    Next i

    ' Close config workbook without saving
    wbConfig.Close SaveChanges:=False
    Application.ScreenUpdating = True

    command = """" & pythonPath & """ """ & scriptPath & """" & args & " > """ & logPath & """ 2>&1"

    Set wsh = CreateObject("WScript.Shell")
    Debug.Print "Command: " & command
    exitCode = wsh.Run("cmd /c """ & command & """", 0, True)

    If exitCode = 0 Then
        MsgBox "Word document generated successfully!", vbInformation, "Done"
    Else
        MsgBox "Script failed. Check log: " & logPath, vbCritical, "Error"
    End If

    Set wsh = Nothing
    Exit Sub

ConfigError:
    Application.ScreenUpdating = True
    MsgBox "Could not open config file: " & configPath, vbCritical, "Config Error"
    Exit Sub

End Sub

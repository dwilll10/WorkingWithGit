def get_excel_value(excel_path, row_value_in_col_c, col_name, sheet_name="Sheet5"):
    # data_only=True returns cached cell values instead of formula strings
    wb = openpyxl.load_workbook(excel_path, data_only=True)
    ws = wb[sheet_name]

    # map each header name to its column index number
    headers = {}
    for col_idx, cell in enumerate(ws[1], start=1):
        if cell.value is not None:
            headers[str(cell.value).strip()] = col_idx

    if col_name not in headers:
        raise ValueError(f"Column '{col_name}' not found. Available: {list(headers.keys())}")

    target_col_idx = headers[col_name]

    # search column C for a matching row label, starting after the header row
    for row in ws.iter_rows(min_row=2):
        col_c_value = row[2].value  # index 2 = column C (0-based)
        if col_c_value is not None and str(col_c_value).strip() == str(row_value_in_col_c).strip():
            value = row[target_col_idx - 1].value  # -1 because row tuple is 0-indexed
            return str(value) if value is not None else ""

    raise ValueError(f"No row found where column C = '{row_value_in_col_c}'")


def replace_in_xml(xml_bytes, placeholder, value):
    # parse the raw XML from the docx
    tree = etree.fromstring(xml_bytes)

    # find all text elements in the Word XML
    for elem in tree.iter(f"{{{W}}}t"):
        if elem.text and placeholder in elem.text:
            elem.text = elem.text.replace(placeholder, value)

    # also handle placeholders split across multiple runs by working on full paragraph text
    for para in tree.iter(f"{{{W}}}p"):
        runs = para.findall(f".//{{{W}}}r")
        full_text = "".join((r.find(f"{{{W}}}t").text or "") for r in runs if r.find(f"{{{W}}}t") is not None)

        if placeholder in full_text:
            new_text = full_text.replace(placeholder, value)
            # put the replaced text in the first run, clear the rest
            text_runs = [r for r in runs if r.find(f"{{{W}}}t") is not None]
            if text_runs:
                text_runs[0].find(f"{{{W}}}t").text = new_text
                for run in text_runs[1:]:
                    run.find(f"{{{W}}}t").text = ""

    return etree.tostring(tree, xml_declaration=True, encoding="UTF-8", standalone=True)


def fill_word_from_excel(word_path, excel_path, replacements, output_path):
    # copy the original docx to the output path so we don't modify the template
    shutil.copy2(word_path, output_path)

    # the parts of the docx that can contain text
    xml_parts = [
        "word/document.xml",
        "word/header1.xml",
        "word/header2.xml",
        "word/header3.xml",
        "word/footer1.xml",
        "word/footer2.xml",
        "word/footer3.xml",
    ]

    # build the full replacement map first
    replacement_map = {}
    for item in replacements:
        placeholder = item["placeholder"]
        try:
            value = get_excel_value(
                excel_path,
                row_value_in_col_c=item["row"],
                col_name=item["col"],
                sheet_name=item.get("sheet", "Sheet5")
            )
            print(f"Replacing {placeholder} → {value}")
            replacement_map[placeholder] = value
        except ValueError as e:
            print(f"WARNING: Skipping {placeholder} — {e}")

    # open the copied docx as a zip and update the XML files in place
    import tempfile
    tmp_path = output_path + ".tmp"

    with zipfile.ZipFile(output_path, "r") as zin:
        with zipfile.ZipFile(tmp_path, "w", zipfile.ZIP_DEFLATED) as zout:
            for item in zin.infolist():
                data = zin.read(item.filename)

                # only process XML parts that contain text
                if item.filename in xml_parts:
                    for placeholder, value in replacement_map.items():
                        data = replace_in_xml(data, placeholder, value)

                zout.writestr(item, data)

    # replace the output file with the updated version
    os.replace(tmp_path, output_path)
    print(f"\nSaved to {output_path}")

def replace_in_paragraph(para, placeholder, value):
    # Word often splits text across multiple runs, so combine them before searching
    full_text = "".join(run.text for run in para.runs)
    if placeholder not in full_text:
        return

    new_text = full_text.replace(placeholder, value)

    if para.runs:
        # write the replaced text into the first run to preserve its formatting
        para.runs[0].text = new_text
        # clear remaining runs so their text doesn't appear twice
        for run in para.runs[1:]:
            run.text = ""


def replace_placeholder_in_doc(doc, placeholder, value):
    for para in doc.paragraphs:
        replace_in_paragraph(para, placeholder, value)

    for table in doc.tables:
        for row in table.rows:
            for cell in row.cells:
                for para in cell.paragraphs:
                    replace_in_paragraph(para, placeholder, value)

    # also check headers and footers in each section
    for section in doc.sections:
        for para in section.header.paragraphs:
            replace_in_paragraph(para, placeholder, value)
        for para in section.footer.paragraphs:
            replace_in_paragraph(para, placeholder, value)

def reorder_sheets(wb, desired_order):
    existing = [s.name for s in wb.sheets]
    order = [name for name in desired_order if name in existing]
    
    if not order:
        logger.info("No sheets from desired_order found in workbook.")
        return
    
    logger.info(f"Reordering. Before: {[s.name for s in wb.sheets]}")
    
    # Grab a stable COM reference to each sheet BEFORE any moves happen.
    # Using the Sheets collection via the workbook's .api avoids xlwings re-lookups.
    sheet_refs = {s.name: s.api for s in wb.sheets}
    
    # Strategy: move each sheet in reverse order, placing it BEFORE the first sheet.
    # After the loop, they end up in the correct forward order at the front of the workbook.
    # Sheets not in `order` drift to the end naturally.
    try:
        for name in reversed(order):
            sheet_api = sheet_refs[name]
            # Re-fetch the first sheet each time — it changes as we move things
            first_sheet_api = wb.sheets[0].api
            if sheet_api is not first_sheet_api:
                sheet_api.Move(Before=first_sheet_api)
    except Exception as e:
        logger.error(f"Failed during reorder: {e}")
        import traceback
        logger.error(traceback.format_exc())
        return
    
    logger.info(f"Reordering complete. After: {[s.name for s in wb.sheets]}")

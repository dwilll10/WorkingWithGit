import os
import sys
import zipfile
import shutil
from lxml import etree

W = "http://schemas.openxmlformats.org/wordprocessingml/2006/main"

XML_PARTS = [
    "word/document.xml",
    "word/header1.xml",
    "word/header2.xml",
    "word/header3.xml",
    "word/footer1.xml",
    "word/footer2.xml",
    "word/footer3.xml",
]


def format_value(value):
    """Convert negative numbers to parentheses format and add commas."""
    try:
        num = float(value.replace(",", ""))
        if num < 0:
            return "(" + f"{abs(num):,.0f}" + ")"
        else:
            return f"{num:,.0f}"
    except ValueError:
        return value


def parse_args(argv):
    """
    argv[1] = word_template_path
    argv[2] = output_path
    argv[3..] = placeholder=value pairs
    """
    if len(argv) < 3:
        print("Usage: WordExportOISOS.py <word_template> <output_path> [placeholder=value ...]")
        sys.exit(1)

    word_path   = argv[1]
    output_path = argv[2]
    replacements = {}

    for arg in argv[3:]:
        if "=" in arg:
            placeholder, value = arg.split("=", 1)
            replacements[placeholder.strip()] = format_value(value.strip())

    return word_path, output_path, replacements


def replace_in_xml(xml_bytes, replacements):
    """
    Parse XML once, apply ALL replacements, return bytes.
    Handles both single-run and split-run placeholder cases.
    """
    # Parse — strip any existing XML declaration to avoid double-declaration issues
    tree = etree.fromstring(xml_bytes)

    for placeholder, value in replacements.items():

        # Pass 1: single-run — placeholder fully inside one <w:t>
        for elem in tree.iter(f"{{{W}}}t"):
            if elem.text and placeholder in elem.text:
                elem.text = elem.text.replace(placeholder, value)

        # Pass 2: split-run — placeholder spans multiple <w:r> in a paragraph
        for para in tree.iter(f"{{{W}}}p"):
            runs = para.findall(f".//{{{W}}}r")
            text_runs = [r for r in runs if r.find(f"{{{W}}}t") is not None]
            if not text_runs:
                continue
            full_text = "".join((r.find(f"{{{W}}}t").text or "") for r in text_runs)
            if placeholder in full_text:
                new_text = full_text.replace(placeholder, value)
                text_runs[0].find(f"{{{W}}}t").text = new_text
                for run in text_runs[1:]:
                    run.find(f"{{{W}}}t").text = ""

    # Serialize back — use encoding="unicode" to get a str, then encode to bytes
    # This avoids lxml emitting a redundant/malformed XML declaration
    xml_str = etree.tostring(tree, encoding="unicode")
    return xml_str.encode("utf-8")


def fill_word(word_path, replacements, output_path):
    """Copy template and replace all placeholders in one pass per XML part."""
    shutil.copy2(word_path, output_path)

    tmp_path = output_path + ".tmp"
    with zipfile.ZipFile(output_path, "r") as zin:
        with zipfile.ZipFile(tmp_path, "w", zipfile.ZIP_DEFLATED) as zout:
            for item in zin.infolist():
                data = zin.read(item.filename)
                if item.filename in XML_PARTS:
                    try:
                        data = replace_in_xml(data, replacements)
                    except Exception as e:
                        print(f"  WARNING: could not process {item.filename}: {e}")
                zout.writestr(item, data)

    os.replace(tmp_path, output_path)
    print(f"\nSaved to {output_path}")
    print("Automation was successfully completed without errors")


if __name__ == "__main__":
    word_path, output_path, replacements = parse_args(sys.argv)

    print(f"Template: {word_path}")
    print(f"Output:   {output_path}")
    print(f"Replacements ({len(replacements)}):")
    for p, v in replacements.items():
        print(f"  {p} = {v}")

    fill_word(word_path, replacements, output_path)

import os

import zipfile

from lxml import etree

BASE_DIR = os.path.join(os.environ["OneDrive"], "Documents")
W = "http://schemas.openxmlformats.org/wordprocessingml/2006/main"


def format_value(value):
    # convert negative numbers to parentheses format and add commas
    try:
        num = float(value.replace(",", ""))
        if num < 0:
            return "(" + f"{abs(num):,.0f}" + ")"
        else:
            return f"{num:,.0f}"
    except ValueError:
        return value


def parse_args(argv):
    # parse placeholder=value pairs passed in from VBA
    replacements = {}
    for arg in argv[1:]:
        if "=" in arg:
            placeholder, value = arg.split("=", 1)
            replacements[placeholder.strip()] = format_value(value.strip())
    return replacements


def replace_in_xml(xml_bytes, placeholder, value):
    tree = etree.fromstring(xml_bytes)

    for elem in tree.iter(f"{{{W}}}t"):
        if elem.text and placeholder in elem.text:
            elem.text = elem.text.replace(placeholder, value)

    # handle placeholders split across multiple runs in a paragraph
    for para in tree.iter(f"{{{W}}}p"):
        runs = para.findall(f".//{{{W}}}r")
        full_text = "".join((r.find(f"{{{W}}}t").text or "") for r in runs if r.find(f"{{{W}}}t") is not None)

        if placeholder in full_text:
            new_text = full_text.replace(placeholder, value)
            text_runs = [r for r in runs if r.find(f"{{{W}}}t") is not None]
            if text_runs:
                text_runs[0].find(f"{{{W}}}t").text = new_text
                for run in text_runs[1:]:
                    run.find(f"{{{W}}}t").text = ""

    return etree.tostring(tree, xml_declaration=True, encoding="UTF-8", standalone=True)


def fill_word_from_values(word_path, replacements, output_path):
    shutil.copy2(word_path, output_path)

    xml_parts = [
        "word/document.xml",
        "word/header1.xml",
        "word/header2.xml",
        "word/header3.xml",
        "word/footer1.xml",
        "word/footer2.xml",
        "word/footer3.xml",
    ]

    tmp_path = output_path + ".tmp"

    with zipfile.ZipFile(output_path, "r") as zin:
        with zipfile.ZipFile(tmp_path, "w", zipfile.ZIP_DEFLATED) as zout:
            for item in zin.infolist():
                data = zin.read(item.filename)
                if item.filename in xml_parts:
                    for placeholder, value in replacements.items():
                        data = replace_in_xml(data, placeholder, value)
                zout.writestr(item, data)

    os.replace(tmp_path, output_path)
    print(f"\nSaved to {output_path}")


if __name__ == "__main__":
    # parse placeholder=value pairs from command line arguments passed in from VBA
    replacements = parse_args(sys.argv)

    for placeholder, value in replacements.items():
        print(f"{placeholder} = {value}")

    fill_word_from_values(
        word_path=os.path.join(BASE_DIR, "File.docx"),
        replacements=replacements,
        output_path=os.path.join(BASE_DIR, "filled_output.docx")
    )

' helper function to extract a value from a simple json string by key
Function ExtractJsonValue(jsonText As String, key As String) As String
    Dim searchKey As String
    Dim startPos As Long
    Dim endPos As Long
    
    searchKey = """" & key & """:"
    startPos = InStr(jsonText, searchKey)
    If startPos = 0 Then
        ExtractJsonValue = ""
        Exit Function
    End If
    
    ' move past the key and colon
    startPos = startPos + Len(searchKey)
    
    ' skip any spaces
    Do While Mid(jsonText, startPos, 1) = " "
        startPos = startPos + 1
    Loop
    
    ' check if value is a string (starts with quote) or number
    If Mid(jsonText, startPos, 1) = """" Then
        ' string value - find closing quote
        startPos = startPos + 1
        endPos = InStr(startPos, jsonText, """")
        ExtractJsonValue = Mid(jsonText, startPos, endPos - startPos)
    Else
        ' number value - find next comma, } or newline
        endPos = startPos
        Do While endPos <= Len(jsonText)
            Dim c As String
            c = Mid(jsonText, endPos, 1)
            If c = "," Or c = "}" Or c = vbLf Or c = vbCr Then Exit Do
            endPos = endPos + 1
        Loop
        ExtractJsonValue = Trim(Mid(jsonText, startPos, endPos - startPos))
    End If
End Function

ElseIf sheetName = "FN22" Then
                Dim phF_FN22 As String
                phF_FN22 = Trim(wsConfig.Cells(i, 3).Value)

                ' Find section in col A
                Dim foundCell22 As Range
                Set foundCell22 = wsData.Columns("A").Find( _
                    What:=section, LookIn:=xlValues, LookAt:=xlWhole, MatchCase:=True)

                If foundCell22 Is Nothing Then
                    Debug.Print "  FN22 NOT FOUND section: '" & section & "'"
                    GoTo NextRow
                End If

                Dim sectionRow22 As Long
                Dim nextSectionRow22 As Long
                Dim nextCell22 As Range
                sectionRow22 = foundCell22.Row

                Set nextCell22 = wsData.Columns("A").FindNext(foundCell22)
                If nextCell22 Is Nothing Or nextCell22.Row <= sectionRow22 Then
                    nextSectionRow22 = wsData.Cells(wsData.Rows.Count, "D").End(xlUp).Row
                Else
                    nextSectionRow22 = nextCell22.Row - 1
                End If

                ' Find subtitle in col D within bounded range
                Dim subtitleRow22 As Long
                Dim sk22 As Long
                subtitleRow22 = 0

                For sk22 = sectionRow22 To nextSectionRow22
                    If Trim(CStr(wsData.Cells(sk22, 4).Value)) = Trim(subtitle) Then
                        subtitleRow22 = sk22
                        Exit For
                    End If
                Next sk22

                If subtitleRow22 = 0 Then
                    Debug.Print "  FN22 NOT FOUND subtitle: '" & subtitle & "' under section: '" & section & "'"
                    GoTo NextRow
                End If

                Debug.Print "FN22 row " & i & ": subtitle='" & subtitle & "' foundRow=" & subtitleRow22

                If phF_FN22 <> "" Then args = args & " """ & phF_FN22 & "=" & Replace(Replace(Trim(wsData.Cells(subtitleRow22, 6).Value), Chr(10), ""), Chr(13), "") & """"

' Load skipped/replaced sheets from SkippedSheets tab in config file
Dim wsSkipped As Worksheet
Dim skippedLastRow As Long

On Error Resume Next
Set wsSkipped = wbConfig.Sheets("SkippedSheets")
On Error GoTo 0

If wsSkipped Is Nothing Then
    MsgBox "Could not find 'SkippedSheets' tab in config file", vbCritical, "Error"
    Exit Sub
End If

skippedLastRow = wsSkipped.Cells(wsSkipped.Rows.Count, 1).End(xlUp).Row

For s = 2 To skippedLastRow
    Dim skipVal As String
    Dim replaceVal As String
    skipVal = LCase(Trim(Replace(CStr(wsSkipped.Cells(s, 1).Value), Chr(160), " ")))
    replaceVal = LCase(Trim(Replace(CStr(wsSkipped.Cells(s, 2).Value), Chr(160), " ")))

    If skipVal <> "" Then
        If replaceVal = "" Then
            ' No replacement — just remove from master list
            If masterTitles.Exists(skipVal) Then masterTitles.Remove skipVal
        Else
            ' Has replacement — remove the original and add the replacement if not already there
            If masterTitles.Exists(skipVal) Then masterTitles.Remove skipVal
            If Not masterTitles.Exists(replaceVal) Then
                masterTitles.Add replaceVal, replaceVal
            End If
        End If
    End If
Next s

' Read subtitle column from col G — blank defaults to C (3)
            Dim sumSubtitleCol As Long
            sumSubtitleCol = 3
            Dim sumSubtitleColStr As String
            sumSubtitleColStr = Trim(CStr(wsSums.Cells(sRow, 7).Value))
            If sumSubtitleColStr <> "" Then
                sumSubtitleCol = ColLetterToNum(sumSubtitleColStr)
            End If

            If sumSection <> "" Then
                ' Has section — find section then subtitle within it
                Dim sumSectionRow As Long
                sumSectionRow = FindSectionRow(sumDataSheet, sumSection)
                If sumSectionRow = 0 Then
                    Debug.Print "Sums: section '" & sumSection & "' not found in " & sumSheet
                    GoTo NextSumRow
                End If
                Dim sumNextSectionRow As Long
                sumNextSectionRow = FindNextSectionRow(sumDataSheet, sumSectionRow, _
                    sumDataSheet.Cells(sumDataSheet.Rows.Count, sumSubtitleCol).End(xlUp).Row)
                sumDataRow = FindSubtitleRow(sumDataSheet, sumSectionRow, sumNextSectionRow, _
                                             sumSubtitle, sumSubtitleCol, sumOccurrence)
            Else
                ' No section — search entire subtitle column
                sumDataRow = FindSubtitleRow(sumDataSheet, 1, _
                    sumDataSheet.Cells(sumDataSheet.Rows.Count, sumSubtitleCol).End(xlUp).Row, _
                    sumSubtitle, sumSubtitleCol, sumOccurrence)
            End If

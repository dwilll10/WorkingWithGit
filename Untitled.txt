' Find pending FS list
            If InStr(1, logLine, "Pending FS:", vbTextCompare) > 0 Then
                Dim pfsSearch As Long
                Dim pfsStart As Long
                Dim pfsEnd As Long
                pfsSearch = InStr(1, logLine, "Pending FS:", vbTextCompare)
                pfsStart = InStr(pfsSearch, logLine, "[") + 1
                pfsEnd = InStr(pfsStart, logLine, "]") - 1
                If pfsStart > 1 And pfsEnd > pfsStart Then
                    pendingFS = Mid(logLine, pfsStart, pfsEnd - pfsStart + 1)
                End If
            End If

' Combine pending FN and FS into one list
    Dim combinedPending As String
    If pendingFN <> "" And pendingFS <> "" Then
        combinedPending = pendingFN & "," & pendingFS
    ElseIf pendingFN <> "" Then
        combinedPending = pendingFN
    Else
        combinedPending = pendingFS
    End If

    If combinedPending <> "" And successTabs <> "" Then
        Dim pendingArr() As String
        Dim p As Long
        pendingArr = Split(combinedPending, ",")
        For p = 0 To UBound(pendingArr)
            Dim pendingTab As String
            pendingTab = Trim(pendingArr(p))
            pendingTab = Replace(pendingTab, "'", "")
            pendingTab = Trim(pendingTab)

            ' Check if tab already existed before this run
            Dim tabAlreadyExists As Boolean
            tabAlreadyExists = False
            Dim chkWs As Worksheet
            For Each chkWs In ThisWorkbook.Worksheets
                If LCase(chkWs.Name) = LCase(pendingTab) Then
                    tabAlreadyExists = True
                    Exit For
                End If
            Next chkWs

            ' Only add if in successTabs AND didn't already exist
            If Not tabAlreadyExists Then
                If InStr("," & LCase(Replace(successTabs, " ", "")) & ",", "," & LCase(Replace(pendingTab, " ", "")) & ",") > 0 Then
                    If createdTabs = "" Then
                        createdTabs = pendingTab
                    Else
                        createdTabs = createdTabs & ", " & pendingTab
                    End If
                End If
            End If
        Next p
    End If

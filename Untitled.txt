'--------------------------------------------------------------------------
    ' Count discrepancies in STMT to STMT tab
    '--------------------------------------------------------------------------
    Dim stmtDiscrepancies As Long
    stmtDiscrepancies = 0
    
    Dim wsStmt As Worksheet
    Dim stmtExists As Boolean
    stmtExists = False
    
    Dim checkWs As Worksheet
    For Each checkWs In ThisWorkbook.Worksheets
        If checkWs.Name = "STMT to STMT" Then
            stmtExists = True
            Set wsStmt = checkWs
            Exit For
        End If
    Next checkWs
    
    If stmtExists Then
        ' Find the Difference column header
        Dim diffCol As Long
        diffCol = 0
        Dim hdrRow As Long
        Dim hdrCol As Long
        For hdrRow = 1 To 10  ' search first 10 rows for header
            For hdrCol = 1 To 50  ' search first 50 columns
                If Trim(CStr(wsStmt.Cells(hdrRow, hdrCol).Value)) = "Difference" Then
                    diffCol = hdrCol
                    Exit For
                End If
            Next hdrCol
            If diffCol > 0 Then Exit For
        Next hdrRow
        
        If diffCol > 0 Then
            Dim lastStmtRow As Long
            lastStmtRow = wsStmt.Cells(wsStmt.Rows.Count, diffCol).End(xlUp).Row
            Dim r As Long
            For r = hdrRow + 1 To lastStmtRow
                Dim cellVal As Variant
                cellVal = wsStmt.Cells(r, diffCol).Value
                If Not IsEmpty(cellVal) And Not IsNull(cellVal) Then
                    If IsNumeric(cellVal) Then
                        If CDbl(cellVal) <> 0 Then
                            stmtDiscrepancies = stmtDiscrepancies + 1
                        End If
                    End If
                End If
            Next r
        Else
            stmtDiscrepancies = -1  ' flag that column wasn't found
        End If
    End If


' Statement to Statement discrepancies
    Dim stmtMsg As String
    If stmtDiscrepancies = -1 Then
        stmtMsg = vbNewLine & "Discrepancies in Statement to Statement: Column 'Difference' not found"
    ElseIf Not stmtExists Then
        stmtMsg = vbNewLine & "Discrepancies in Statement to Statement: Tab 'STMT to STMT' not found"
    Else
        stmtMsg = vbNewLine & "Discrepancies in Statement to Statement: " & stmtDiscrepancies
    End If

def parse_args(argv):
    if len(argv) < 3:
        print("Usage: WordExportOISOS.py <word_template> <output_path> <args_file>")
        sys.exit(1)

    word_path   = argv[1]
    output_path = argv[2]
    args_file   = argv[3]
    replacements = {}

    with open(args_file, 'r') as f:
        args_string = f.read()

    # Parse the args string — each pair is wrapped in quotes
    import re
    pairs = re.findall(r'"([^"]+)"', args_string)
    for pair in pairs:
        if "=" in pair:
            placeholder, value = pair.split("=", 1)
            replacements[placeholder.strip()] = format_value(value.strip())

    return word_path, output_path, replacements

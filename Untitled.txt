def fill_word(word_path, replacements, output_path):
    if os.path.exists(output_path):
        os.remove(output_path)

    with zipfile.ZipFile(word_path, "r") as zin:
        with zipfile.ZipFile(output_path, "w", zipfile.ZIP_DEFLATED) as zout:
            for item in zin.infolist():
                data = zin.read(item.filename)
                if item.filename in XML_PARTS:
                    for placeholder, value in replacements.items():
                        data = replace_in_xml(data, placeholder, value)
                zout.writestr(item, data)

    # Validate output zip
    try:
        with zipfile.ZipFile(output_path, "r") as z:
            bad = z.testzip()
            if bad:
                print(f"ZIP ERROR: first bad file is {bad}")
            else:
                print(f"ZIP OK: {len(z.namelist())} files, size={os.path.getsize(output_path)}")
    except Exception as e:
        print(f"ZIP OPEN FAILED: {e}")

    print(f"\nSaved to {output_path}")
    print("Automation was successfully completed without errors")

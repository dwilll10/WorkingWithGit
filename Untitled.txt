Dim wsMaster As Worksheet
Dim wbMaster As Workbook
Dim masterPath As String
Dim masterLastRow As Long
Dim masterTitleCol As Long
Dim m As Long
Dim headerCell As Range

masterPath = oneDrivePath & "\Documents\z. Python Program\GK_new_format\BSFNMasterList.xlsx"  ' <-- adjust path as needed

On Error GoTo MasterError
Set wbMaster = Workbooks.Open(masterPath, ReadOnly:=True)
On Error GoTo 0

Set wsMaster = wbMaster.Sheets(1)

' Find the "Title" column by header name
Set headerCell = wsMaster.Rows(1).Find( _
    What:="Title", LookIn:=xlValues, LookAt:=xlWhole, MatchCase:=False)

If headerCell Is Nothing Then
    wbMaster.Close SaveChanges:=False
    MsgBox "Could not find 'Title' column in BSFNMasterList", vbCritical, "Error"
    Exit Sub
End If

masterTitleCol = headerCell.Column
masterLastRow = wsMaster.Cells(wsMaster.Rows.Count, masterTitleCol).End(xlUp).Row

For m = 2 To masterLastRow
    If existingSheets = "" Then
        existingSheets = LCase(Trim(wsMaster.Cells(m, masterTitleCol).Value))
    Else
        existingSheets = existingSheets & "," & LCase(Trim(wsMaster.Cells(m, masterTitleCol).Value))
    End If
Next m

wbMaster.Close SaveChanges:=False



MasterError:
    Application.ScreenUpdating = True
    MsgBox "Could not open BSFNMasterList file: " & masterPath, vbCritical, "Master List Error"
    Exit Sub

' Load all master list titles into a dictionary before the loop
Dim masterTitles As Object
Set masterTitles = CreateObject("Scripting.Dictionary")
For m = 2 To masterLastRow
    Dim masterTitle As String
    masterTitle = LCase(Trim(wsMaster.Cells(m, masterTitleCol).Value))
    If masterTitle <> "" And Not masterTitles.Exists(masterTitle) Then
        masterTitles.Add masterTitle, wsMaster.Cells(m, masterTitleCol).Value  ' store original casing as value
    End If
Next m

If dataExists Then
            masterTitles.Remove LCase(Trim(sheetName))
        End If

' Report master list entries with no matching config tab
    If masterTitles.Count > 0 Then
        Dim unmatchedList As String
        unmatchedList = ""
        Dim key As Variant
        For Each key In masterTitles.Keys
            If unmatchedList = "" Then
                unmatchedList = masterTitles(key)
            Else
                unmatchedList = unmatchedList & ", " & masterTitles(key)
            End If
        Next key
        WriteLog "Master list titles with no matching config tab: " & unmatchedList
        MsgBox "The following master list entries had no matching config tab:" & vbNewLine & unmatchedList, vbExclamation, "Unmatched Master List Entries"
    End If
